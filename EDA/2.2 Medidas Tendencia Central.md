SELECT
    'discounted_price' AS variable,
    MIN(discounted_price) AS min,
    MAX(discounted_price) AS max,
    APPROX_QUANTILES(discounted_price, 2)[OFFSET(1)] AS median,
    APPROX_TOP_COUNT(discounted_price, 1)[OFFSET(0)].value AS mode,
    STDDEV(discounted_price) AS stddev
  FROM
    `datalab-433117.dataset.tabla_unificada`
UNION ALL
SELECT
    'actual_price' AS variable,

    `datalab-433117.dataset.tabla_unificada`
UNION ALL
SELECT

    APPROX_TOP_COUNT(discount_percentage, 1)[OFFSET(0)].value AS mode,

    `datalab-433117.dataset.tabla_unificada`
UNION ALL

  FROM
    `datalab-433117.dataset.tabla_unificada`
